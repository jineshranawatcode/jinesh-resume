This demo showcases advanced Delta Lake capabilities within Azure Databricks, a critical component of modern data engineering practices. Delta Lake provides ACID transactions, schema enforcement, time travel, and optimized storage, solving key challenges in data lake architectures.

1. Setting Up Delta Lake Environment

# Configure Spark to use Delta Lake
from pyspark.sql import SparkSession
from delta.tables import *
import pyspark.sql.functions as F

# Create Spark session with Delta Lake support
spark = SparkSession.builder \
    .appName("DeltaLakeDemo") \
    .config("spark.sql.extensions", "io.delta.sql.DeltaSparkSessionExtension") \
    .config("spark.sql.catalog.spark_catalog", "org.apache.spark.sql.delta.catalog.DeltaCatalog") \
    .config("spark.databricks.delta.schema.autoMerge.enabled", "true") \
    .getOrCreate()

# Define data path (using DBFS in Databricks)
delta_path = "/mnt/data/customer_orders_delta"

print("Delta Lake environment configured successfully!")

This initialization code sets up the Spark session with the necessary Delta Lake configurations, including schema auto-merging capability which simplifies schema evolution.

2. Creating a Delta Table with Complex Schema

# Generate sample data with nested structure
from pyspark.sql.types import *
import datetime
import random

# Define schema with nested structures
customer_schema = StructType([
    StructField("customer_id", StringType(), False),
    StructField("name", StringType(), True),
    StructField("email", StringType(), True),
    StructField("registration_date", TimestampType(), True),
    StructField("address", StructType([
        StructField("street", StringType(), True),
        StructField("city", StringType(), True),
        StructField("state", StringType(), True),
        StructField("zip", StringType(), True)
    ]), True),
    StructField("orders", ArrayType(StructType([
        StructField("order_id", StringType(), True),
        StructField("order_date", TimestampType(), True),
        StructField("amount", DoubleType(), True),
        StructField("items", ArrayType(StructType([
            StructField("product_id", StringType(), True),
            StructField("quantity", IntegerType(), True),
            StructField("price", DoubleType(), True)
        ])), True)
    ])), True)
])

# Create sample data
data = []
cities = ["New York", "Los Angeles", "Chicago", "Houston", "Phoenix"]
states = ["NY", "CA", "IL", "TX", "AZ"]
products = ["P1001", "P1002", "P1003", "P1004", "P1005"]

for i in range(1, 101):
    customer_id = f"C{1000 + i}"
    city_idx = random.randint(0, 4)
    
    orders = []
    num_orders = random.randint(1, 5)
    for j in range(1, num_orders + 1):
        order_id = f"ORD-{customer_id}-{j}"
        order_date = datetime.datetime.now() - datetime.timedelta(days=random.randint(1, 365))
        
        items = []
        num_items = random.randint(1, 4)
        order_amount = 0
        for k in range(num_items):
            product_id = products[random.randint(0, 4)]
            quantity = random.randint(1, 5)
            price = round(random.uniform(10, 100), 2)
            item_total = quantity * price
            order_amount += item_total
            
            items.append({
                "product_id": product_id,
                "quantity": quantity,
                "price": price
            })
        
        orders.append({
            "order_id": order_id,
            "order_date": order_date,
            "amount": round(order_amount, 2),
            "items": items
        })
    
    customer = {
        "customer_id": customer_id,
        "name": f"Customer {i}",
        "email": f"customer{i}@example.com",
        "registration_date": datetime.datetime.now() - datetime.timedelta(days=random.randint(30, 730)),
        "address": {
            "street": f"{random.randint(100, 999)} Main St",
            "city": cities[city_idx],
            "state": states[city_idx],
            "zip": f"{random.randint(10000, 99999)}"
        },
        "orders": orders
    }
    
    data.append(customer)

# Create DataFrame and write to Delta Lake
customers_df = spark.createDataFrame(data, customer_schema)
customers_df.write.format("delta").mode("overwrite").save(delta_path)

# Create Delta table
spark.sql(f"CREATE TABLE IF NOT EXISTS customer_orders USING DELTA LOCATION '{delta_path}'")

print(f"Created Delta table with {customers_df.count()} records")
display(spark.sql("DESCRIBE DETAIL customer_orders"))

This code demonstrates creating a complex schema with nested structures (common in real-world scenarios) and generating realistic sample data. The Delta table creation showcases a key advantage of Delta Lake: its ability to handle complex, nested data structures efficiently.

3. Implementing Schema Evolution

# Schema evolution with mergeSchema option
from pyspark.sql.types import StructType, StructField, StringType, DoubleType, ArrayType

# Read existing Delta table
delta_df = spark.read.format("delta").load(delta_path)

# Create sample data with new columns
new_data = [
    {
        "customer_id": "C1001", 
        "loyalty_tier": "GOLD",
        "loyalty_points": 1250,
        "preferences": {
            "preferred_contact": "EMAIL",
            "marketing_opt_in": True,
            "categories": ["Electronics", "Home"]
        }
    },
    {
        "customer_id": "C1002", 
        "loyalty_tier": "SILVER",
        "loyalty_points": 750,
        "preferences": {
            "preferred_contact": "SMS",
            "marketing_opt_in": False,
            "categories": ["Clothing"]
        }
    }
]

# Define schema for new data with additional columns
new_schema = StructType([
    StructField("customer_id", StringType(), False),
    StructField("loyalty_tier", StringType(), True),
    StructField("loyalty_points", DoubleType(), True),
    StructField("preferences", StructType([
        StructField("preferred_contact", StringType(), True),
        StructField("marketing_opt_in", StringType(), True),
        StructField("categories", ArrayType(StringType()), True)
    ]), True)
])

# Create DataFrame with new columns
new_df = spark.createDataFrame(new_data, new_schema)

# Write with mergeSchema option to evolve schema
new_df.write.format("delta") \
    .option("mergeSchema", "true") \
    .mode("append") \
    .save(delta_path)

# Verify schema evolution
evolved_df = spark.read.format("delta").load(delta_path)
print("Schema after evolution:")
evolved_df.printSchema()

# Show data with new columns
print("\nData with new columns:")
display(evolved_df.select("customer_id", "loyalty_tier", "loyalty_points", "preferences").where("loyalty_tier IS NOT NULL"))

This example demonstrates Delta Lake's schema evolution capabilities, allowing you to add new columns to your data model without breaking existing queries - a critical feature for agile data environments.

4. Time Travel and Version Management

# Demonstrate Delta Lake time travel
from delta.tables import DeltaTable
import time

# Get current table version
history = spark.sql(f"DESCRIBE HISTORY delta.`{delta_path}`")
current_version = history.select("version").first()[0]
print(f"Current table version: {current_version}")

# Make some updates
delta_table = DeltaTable.forPath(spark, delta_path)
delta_table.update(
    condition="address.state = 'CA'",
    set={"address.state": F.lit("California")}
)

# Check history after update
history = spark.sql(f"DESCRIBE HISTORY delta.`{delta_path}`")
display(history.select("version", "timestamp", "operation", "operationParameters"))

# Count records in California after update
updated_df = spark.read.format("delta").load(delta_path)
california_count = updated_df.filter("address.state = 'California'").count()
print(f"Records with state 'California' after update: {california_count}")

# Time travel to previous version
previous_df = spark.read.format("delta").option("versionAsOf", current_version).load(delta_path)
ca_count = previous_df.filter("address.state = 'CA'").count()
california_count = previous_df.filter("address.state = 'California'").count()
print(f"At version {current_version}:")
print(f" - Records with state 'CA': {ca_count}")
print(f" - Records with state 'California': {california_count}")

# Demonstrate rollback
delta_table.restoreToVersion(current_version)
print(f"Table rolled back to version {current_version}")

# Verify rollback
restored_df = spark.read.format("delta").load(delta_path)
ca_count = restored_df.filter("address.state = 'CA'").count()
california_count = restored_df.filter("address.state = 'California'").count()
print("After rollback:")
print(f" - Records with state 'CA': {ca_count}")
print(f" - Records with state 'California': {california_count}")

This section demonstrates one of Delta Lake's most powerful features - time travel. The ability to query data at any point in time, review historical changes, and rollback to previous versions is invaluable for data governance, auditing, and recovery scenarios.

5. Optimizing Delta Lake Performance

# Delta Lake optimization techniques
import time

# Run a query and measure performance
def measure_query_time(query_func):
    start_time = time.time()
    result = query_func()
    end_time = time.time()
    return result, (end_time - start_time)

# Query to find high-value customers in New York
def query_high_value_customers():
    return spark.sql("""
        SELECT 
            c.customer_id,
            c.name,
            c.email,
            c.address.city,
            SUM(o.amount) as total_spend
        FROM (
            SELECT 
                customer_id, 
                name, 
                email, 
                address,
                EXPLODE(orders) as o
            FROM delta.`{}`) c
        WHERE c.address.city = 'New York'
        GROUP BY c.customer_id, c.name, c.email, c.address.city
        HAVING SUM(o.amount) > 300
        ORDER BY total_spend DESC
    """.format(delta_path))

# Run query before optimization
print("Running query before optimization...")
result_before, time_before = measure_query_time(query_high_value_customers)
print(f"Query execution time before optimization: {time_before:.2f} seconds")
display(result_before)

# Optimize the Delta table with Z-Order
print("\nOptimizing table...")
spark.sql(f"""
    OPTIMIZE delta.`{delta_path}`
    ZORDER BY (customer_id, `address.city`)
""")

# Check optimization history
optimize_history = spark.sql(f"DESCRIBE HISTORY delta.`{delta_path}`").filter("operation = 'OPTIMIZE'")
display(optimize_history)

# Run the same query after optimization
print("\nRunning query after optimization...")
result_after, time_after = measure_query_time(query_high_value_customers)
print(f"Query execution time after optimization: {time_after:.2f} seconds")
print(f"Performance improvement: {(time_before - time_after) / time_before * 100:.2f}%")

# Vacuum old files (usually set higher in production)
print("\nRunning vacuum to clean up old files...")
# Setting retention to 0 hours for demo purposes
# In production, use longer retention periods (default is 7 days)
spark.sql(f"VACUUM delta.`{delta_path}` RETAIN 0 HOURS")

# Check table details after optimization
table_details = spark.sql(f"DESCRIBE DETAIL delta.`{delta_path}`")
display(table_details)

This section demonstrates Delta Lake's performance optimization capabilities, including Z-Order indexing and vacuuming. Z-Order optimizes data layout for common query patterns, while vacuuming manages file system cleanup - both critical for large-scale data lakes.

spark.sql(f"ALTER TABLE delta.`{delta_path}` SET TBLPROPERTIES (delta.enableChangeDataFeed = true)")

history = spark.sql(f"DESCRIBE HISTORY delta.`{delta_path}`")

# Get end version after changes
history = spark.sql(f"DESCRIBE HISTORY delta.`{delta_path}`")

    "_change_type",

    "name",

    "_commit_version"
))

This section demonstrates Delta Lake's Change Data Feed (CDF) feature, which enables efficient change data capture for downstream processing, a critical capability for data integration and real-time data pipelines.

Executable Python Example: Real-time Data Processing with Delta Tables

This Python example demonstrates how to implement a real-time data processing pipeline using Delta Lake. This code can be run locally on your machine with the proper libraries installed. Created by Jinesh Ranawat, this example showcases key Delta Lake concepts including streaming writes, schema evolution, and compaction.

Prerequisites

To run this example locally, you'll need:

  • Python 3.8+ installed
  • Required libraries: pyspark, delta-spark, pandas, numpy, faker
  • Installation: pip install pyspark==3.4.0 delta-spark==2.4.0 pandas numpy faker
# real_time_delta_processor.py
# A complete local Python script for real-time data processing with Delta Lake
# Created by Jinesh Ranawat

import os
import time
import random
import shutil
import pandas as pd
import numpy as np
from datetime import datetime, timedelta
from faker import Faker
from pyspark.sql import SparkSession
from pyspark.sql.types import *
from pyspark.sql.functions import *
from delta.tables import *

# Configuration
LOCAL_DELTA_PATH = "./delta_tables"
LOCAL_CHECKPOINT_PATH = "./checkpoints"
STREAM_DATA_PATH = "./stream_data"
BATCH_SIZE = 100
NUM_BATCHES = 5
STREAM_INTERVAL_SECONDS = 2

# Clean up previous runs
for path in [LOCAL_DELTA_PATH, LOCAL_CHECKPOINT_PATH, STREAM_DATA_PATH]:
    if os.path.exists(path):
        shutil.rmtree(path)
    os.makedirs(path)

# Initialize Faker for generating realistic data
fake = Faker()

# Create Spark session with Delta support
def create_spark_session():
    return (SparkSession.builder
            .appName("DeltaLakeRealTimeProcessing")
            .config("spark.sql.extensions", "io.delta.sql.DeltaSparkSessionExtension")
            .config("spark.sql.catalog.spark_catalog", "org.apache.spark.sql.delta.catalog.DeltaCatalog")
            .config("spark.databricks.delta.schema.autoMerge.enabled", "true")
            .config("spark.sql.shuffle.partitions", "1") # For local demo only
            .master("local[*]")
            .getOrCreate())

spark = create_spark_session()
print("Spark session created successfully")

# Define schema for the transaction data
transaction_schema = StructType([
    StructField("transaction_id", StringType(), False),
    StructField("customer_id", StringType(), True),
    StructField("transaction_time", TimestampType(), True),
    StructField("amount", DoubleType(), True),
    StructField("product_id", StringType(), True),
    StructField("store_id", StringType(), True),
    StructField("payment_method", StringType(), True),
    StructField("is_approved", BooleanType(), True)
])

# Generate transaction data
def generate_transaction_data(batch_id, batch_size):
    transactions = []
    
    for i in range(batch_size):
        transaction_id = f"T-{batch_id}-{i}"
        customer_id = f"C{random.randint(1000, 9999)}"
        transaction_time = datetime.now() - timedelta(seconds=random.randint(0, 60))
        amount = round(random.uniform(10.0, 500.0), 2)
        product_id = f"P{random.randint(100, 999)}"
        store_id = f"S{random.randint(10, 99)}"
        payment_method = random.choice(["CREDIT", "DEBIT", "CASH", "MOBILE"])
        is_approved = random.random() > 0.05  # 5% rejected transactions
        
        transactions.append({
            "transaction_id": transaction_id,
            "customer_id": customer_id,
            "transaction_time": transaction_time,
            "amount": amount,
            "product_id": product_id,
            "store_id": store_id,
            "payment_method": payment_method,
            "is_approved": is_approved
        })
    
    return transactions

# Generate and save transaction data as JSON for streaming
def generate_stream_data():
    print("Generating streaming data...")
    os.makedirs(f"{STREAM_DATA_PATH}/transactions", exist_ok=True)
    
    for batch_id in range(1, NUM_BATCHES + 1):
        transactions = generate_transaction_data(batch_id, BATCH_SIZE)
        df = pd.DataFrame(transactions)
        
        # Convert timestamps to string for JSON serialization
        df['transaction_time'] = df['transaction_time'].astype(str)
        
        # Save as JSON file
        file_path = f"{STREAM_DATA_PATH}/transactions/batch_{batch_id}.json"
        df.to_json(file_path, orient='records', lines=True)
        
        print(f"Generated batch {batch_id} with {BATCH_SIZE} transactions")
        time.sleep(STREAM_INTERVAL_SECONDS)

# Create customer data with additional details
def create_customer_data():
    customers = []
    
    for i in range(1000, 10000):
        customer_id = f"C{i}"
        
        customers.append({
            "customer_id": customer_id,

            "loyalty_tier": random.choice(["BRONZE", "SILVER", "GOLD", "PLATINUM"]),

        })
    
    # Convert to DataFrame and write to Delta
    customer_df = spark.createDataFrame(customers)
    customer_df.write.format("delta").mode("overwrite").save(f"{LOCAL_DELTA_PATH}/customers")
    
    print(f"Created customer Delta table with {len(customers)} records")
    return customer_df

# Create product data
def create_product_data():
    products = []
    categories = ["Electronics", "Clothing", "Home", "Food", "Beauty", "Sports"]
    
    for i in range(100, 1000):
        product_id = f"P{i}"
        category = random.choice(categories)
        
        products.append({
            "product_id": product_id,

    print(f"Created product Delta table with {len(products)} records")
    return product_df

# Setup streaming query to process transaction data
def setup_streaming_query():
    print("Setting up streaming query...")
    
    # Read the stream
    transaction_stream = (
        spark.readStream
        .schema(transaction_schema)
        .json(f"{STREAM_DATA_PATH}/transactions")
    )
    
    # Process the stream with watermarking and windowing
    processed_stream = (
        transaction_stream
        .withWatermark("transaction_time", "1 minute")
        .withColumn("processing_time", current_timestamp())
        .withColumn("hour_window", window("transaction_time", "1 hour"))
    )
    
    # Write to Delta table with checkpointing
    stream_query = (
        processed_stream
        .writeStream
        .format("delta")
        .option("checkpointLocation", f"{LOCAL_CHECKPOINT_PATH}/transactions")
        .outputMode("append")
        .start(f"{LOCAL_DELTA_PATH}/transactions")
    )
    
    return stream_query

# Create aggregated sales metrics in Delta table
def create_sales_metrics():
    # Wait for some streaming data to arrive
    time.sleep(STREAM_INTERVAL_SECONDS * 2)
    
    # Define the Delta tables
    transactions_delta = DeltaTable.forPath(spark, f"{LOCAL_DELTA_PATH}/transactions")
    
    # Create a view for SQL
    transactions_delta.toDF().createOrReplaceTempView("transactions")
    
    # Create sales metrics with SQL and write to Delta
    sales_metrics = spark.sql("""
        SELECT 
            product_id,
            store_id,
            COUNT(*) as transaction_count,
            SUM(amount) as total_sales,
            AVG(amount) as avg_transaction_value,
            COUNT(CASE WHEN is_approved = false THEN 1 END) as rejected_count
        FROM transactions
        GROUP BY product_id, store_id
    """)
    
    sales_metrics.write.format("delta").mode("overwrite").save(f"{LOCAL_DELTA_PATH}/sales_metrics")
    print("Created sales metrics Delta table")
    return sales_metrics

# Demonstrate schema evolution by adding columns
def demonstrate_schema_evolution():
    print("\nDemonstrating schema evolution...")
    
    # Get the current schema
    current_df = spark.read.format("delta").load(f"{LOCAL_DELTA_PATH}/transactions")
    print("Original schema:")
    current_df.printSchema()
    
    # Add new fields to the schema
    new_data = spark.createDataFrame([
        {
            "transaction_id": f"T-EVO-1",
            "customer_id": "C5000",
            "transaction_time": datetime.now(),
            "amount": 150.0,
            "product_id": "P500",
            "store_id": "S25",
            "payment_method": "CRYPTO",
            "is_approved": True,
            "promotion_code": "SUMMER25",
            "discount_amount": 25.0,
            "loyalty_points_earned": 150
        }
    ])
    
    # Write with schema evolution
    new_data.write.format("delta").mode("append").option("mergeSchema", "true").save(f"{LOCAL_DELTA_PATH}/transactions")
    
    # Check updated schema
    evolved_df = spark.read.format("delta").load(f"{LOCAL_DELTA_PATH}/transactions")
    print("\nEvolved schema:")
    evolved_df.printSchema()
    
    # Verify the new row is there
    new_row = evolved_df.filter("transaction_id = 'T-EVO-1'")
    print("\nNew row with extended schema:")
    new_row.show(truncate=False)

# Demonstrate Delta time travel
def demonstrate_time_travel():
    print("\nDemonstrating Delta time travel...")
    
    # Get the current version
    history = spark.sql(f"DESCRIBE HISTORY delta.`{LOCAL_DELTA_PATH}/transactions`")
    versions = history.select("version").collect()
    
    if len(versions) > 1:
        current_version = versions[0][0]
        previous_version = versions[1][0]
        
        # Count records in current version
        current_count = spark.read.format("delta").load(f"{LOCAL_DELTA_PATH}/transactions").count()
        print(f"Current version {current_version} record count: {current_count}")
        
        # Count records in previous version
        previous_count = spark.read.format("delta").option("versionAsOf", previous_version).load(f"{LOCAL_DELTA_PATH}/transactions").count()
        print(f"Previous version {previous_version} record count: {previous_count}")
        
        # Show history
        print("\nTable history:")
        history.select("version", "timestamp", "operation", "operationParameters").show(truncate=False)
    else:
        print("Not enough versions available for time travel demonstration")

# Demonstrate optimization and vacuuming
def demonstrate_optimization():
    print("\nDemonstrating Delta optimization...")
    
    # Optimize table
    spark.sql(f"OPTIMIZE delta.`{LOCAL_DELTA_PATH}/transactions` ZORDER BY (transaction_time, store_id)")
    
    # Show optimize history
    history = spark.sql(f"DESCRIBE HISTORY delta.`{LOCAL_DELTA_PATH}/transactions`")
    optimize_history = history.filter(history.operation == "OPTIMIZE")
    print("Optimization history:")
    optimize_history.show(truncate=False)
    
    # Vacuum
    # Note: Using retention=0 for demo; in production use longer retention
    spark.sql(f"VACUUM delta.`{LOCAL_DELTA_PATH}/transactions` RETAIN 0 HOURS")

# Join Delta tables and perform analysis
def perform_analysis():
    print("\nPerforming data analysis...")
    
    # Load all Delta tables
    transactions = spark.read.format("delta").load(f"{LOCAL_DELTA_PATH}/transactions")
    customers = spark.read.format("delta").load(f"{LOCAL_DELTA_PATH}/customers")
    products = spark.read.format("delta").load(f"{LOCAL_DELTA_PATH}/products")
    
    # Register as views
    transactions.createOrReplaceTempView("transactions")
    customers.createOrReplaceTempView("customers")
    products.createOrReplaceTempView("products")
    
    # Perform analysis
    analysis = spark.sql("""
        SELECT 
            c.loyalty_tier,
            p.category,
            COUNT(t.transaction_id) as num_transactions,
            SUM(t.amount) as total_spend,
            AVG(t.amount) as avg_transaction_value,
            SUM(CASE WHEN t.is_approved = false THEN 1 ELSE 0 END) as rejected_count
        FROM transactions t
        JOIN customers c ON t.customer_id = c.customer_id
        JOIN products p ON t.product_id = p.product_id
        GROUP BY c.loyalty_tier, p.category
        ORDER BY c.loyalty_tier, total_spend DESC
    """)
    
    print("\nCustomer spending analysis by loyalty tier and product category:")
    analysis.show(truncate=False)
    
    # Calculate percentage of rejected transactions by payment method
    rejected_analysis = spark.sql("""
        SELECT 
            payment_method,
            COUNT(*) as total_transactions,
            SUM(CASE WHEN is_approved = false THEN 1 ELSE 0 END) as rejected_count,
            (SUM(CASE WHEN is_approved = false THEN 1 ELSE 0 END) / COUNT(*)) * 100 as rejection_rate
        FROM transactions
        GROUP BY payment_method
        ORDER BY rejection_rate DESC
    """)
    
    print("\nRejection rate by payment method:")
    rejected_analysis.show(truncate=False)
    
    # Top 10 customers by spend
    top_customers = spark.sql("""
        SELECT 
            c.customer_id,
            c.name,
            c.loyalty_tier,
            SUM(t.amount) as total_spend,
            COUNT(t.transaction_id) as transaction_count
        FROM transactions t
        JOIN customers c ON t.customer_id = c.customer_id
        WHERE t.is_approved = true
        GROUP BY c.customer_id, c.name, c.loyalty_tier
        ORDER BY total_spend DESC
        LIMIT 10
    """)
    
    print("\nTop 10 customers by spend:")
    top_customers.show(truncate=False)

# Main execution
def main():
    try:
        print("Starting Delta Lake Real-time Processing Demo...")
        
        # Create reference data
        create_customer_data()
        create_product_data()
        
        # Start streaming process
        stream_query = setup_streaming_query()
        
        # Generate streaming data in a separate thread
        import threading
        stream_thread = threading.Thread(target=generate_stream_data)
        stream_thread.start()
        
        # Create sales metrics
        create_sales_metrics()
        
        # Wait for streaming to complete
        stream_thread.join()
        
        # Wait for processing to complete
        stream_query.processAllAvailable()
        
        # Demonstrate Delta features
        demonstrate_schema_evolution()
        demonstrate_time_travel()
        demonstrate_optimization()
        
        # Perform analysis on the data
        perform_analysis()
        
        # Stop the streaming query
        stream_query.stop()
        
        print("\nDemo completed successfully!")
        print(f"Generated Delta tables can be found at: {os.path.abspath(LOCAL_DELTA_PATH)}")
        
    except Exception as e:
        print(f"Error in demo: {str(e)}")
    finally:
        spark.stop()

if __name__ == "__main__":
    main()
